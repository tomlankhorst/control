94  T clip(T u) {
95  if (Limit == std::numeric_limits<T>::infinity()) {
96  return u;
97  }
98 
99  // Determine if the signal is in range
100  if (!clipping && (u > Limit || u < -Limit)) {
101  clipping = true;
102  } else if (clipping && (u >= -Limit && u <= Limit)) {
103  clipping = false;
104  }
105 
106  // If limiting, then clip to min / max
107  if (clipping) {
108  u = std::max(-Limit, std::min(Limit, u));
